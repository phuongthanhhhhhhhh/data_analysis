def toNumeric(dataframe, fromAtt, toAtt):
  from sklearn import preprocessing
  if dataframe[fromAtt].dtype == type(object):
      le = preprocessing.LabelEncoder()# Dummies / One Hot Encoder  
      dataframe[toAtt] = le.fit_transform(dataframe[fromAtt].astype(str))   
  return dataframe

# DBSCan gives us a metric LOFFactor for each data points
# if LOFFactor greater than 3.0 means Outliers or Noises 
# if LOFFactor greater than 2.0 means Border Points and others are Core Points
def CalculateLOFFactor(DataFrame, FeatureList, neighbors = 10):
  import numpy as np
  from sklearn.neighbors import LocalOutlierFactor

  clf = LocalOutlierFactor(n_neighbors=neighbors)
  AttData = DataFrame.loc[:, FeatureList].values
  y_pred = clf.fit_predict(AttData)
  AttData_scores = -1 * clf.negative_outlier_factor_
  LOFFactorData = pd.DataFrame(AttData_scores, columns=['LOF'])
  return LOFFactorData

def ClusterByGaussianMixture(data, FeatureList, nCluster, ClusterColName = 'cluster_GMM'):
  data_train = data.loc[:,FeatureList]

  from sklearn.mixture import GaussianMixture
  Gmm_Model = GaussianMixture(n_components=nCluster, random_state=1234)
  Gmm_Model.fit(data_train)
  predict_label_train = Gmm_Model.fit_predict(data_train)
  data[ClusterColName] = predict_label_train
  
  print(data[ClusterColName].unique())
  
  # ClusterData = pd.DataFrame(columns=FeatureList + ['ClusterNo', 'NumElements'])
  ClusterData_GMM = pd.DataFrame(Gmm_Model.means_)
  ClusterData_GMM.columns = FeatureList
  ClusterData_GMM["ClusterNo"] = ClusterData_GMM.index
  ClusterData_GMM["NumElementRate"] = Gmm_Model.weights_

  return Gmm_Model, ClusterData_GMM, data

def plot_confusion_matrix(cm, classes=['Not Fraud', 'Fraud'],
                          normalize=False,
                          title='Fraud Confusion matrix',
                          cmap=plt.cm.Blues):
    """
    This function prints and plots the confusion matrix.
    Normalization can be applied by setting `normalize=True`.
    From:
        http://scikit-learn.org/stable/auto_examples/model_selection/plot_confusion_matrix.html#sphx-glr-auto-
        examples-model-selection-plot-confusion-matrix-py
    """
    if normalize:
        cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
        print("Normalized confusion matrix")
    else:
        print('Confusion matrix, without normalization')

    # print(cm)

    plt.imshow(cm, interpolation='nearest', cmap=cmap)
    plt.title(title)
    plt.colorbar()
    tick_marks = np.arange(len(classes))
    plt.xticks(tick_marks, classes, rotation=45)
    plt.yticks(tick_marks, classes)

    fmt = '.2f' if normalize else 'd'
    thresh = cm.max() / 2.
    for i, j in product(range(cm.shape[0]), range(cm.shape[1])):
        plt.text(j, i, format(cm[i, j], fmt),
                 horizontalalignment="center",
                 color="white" if cm[i, j] > thresh else "black")

    plt.tight_layout()
    plt.ylabel('True label')
    plt.xlabel('Predicted label')
    plt.show()
